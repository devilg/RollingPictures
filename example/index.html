手风琴式图片效果

通过绝对定位方式实现

通过修改元素width方式实现

配置项

var options = {
	divHeight : 200,
	// 最外层div高度
	divWidth : 500, 
	//最外层div宽度
	normalWidth : 100, 
	// 图片正常显示大小 当divWidth定义时 无作用
	bigWidth : 310, 
	// 图片最大化显示大小
	borderWidth : 2,
	// 图片右侧边框
	transitionTime : 2,
	//动画移动时间
	type : 'left' 
	//展现的方式 'left'通过绝对定位 ，'width'通过修改元素大小
};
